       describe('private-contract SQL schema lint', () => {
         let workdir: string;

         beforeAll(async () => {
           workdir = mkdtempSync(join(tmpdir(), 'schema-lint-'));
           // Apply 0000_v1_baseline.sql against the test DB.
           await execAsync(`pnpm --filter @spatula/db exec tsx src/run-migrate.ts`, {
             env: { ...process.env, DATABASE_URL: databaseUrl },
           });
         }, 60_000);

         afterAll(() => {
           if (workdir) rmSync(workdir, { recursive: true, force: true });
         });

         it('introspected schema matches committed baseline.schema.json', async () => {
           // Use drizzle-kit introspect to produce a JSON snapshot.
           // CLI form: `drizzle-kit introspect --dialect=postgresql --url=$DATABASE_URL --out=$workdir`
           // (validate exact flag names against the installed drizzle-kit version)
           await execAsync(
             `pnpm exec drizzle-kit introspect --dialect=postgresql --out=${workdir}`,
             { env: { ...process.env, DATABASE_URL: databaseUrl } },
           );

           // drizzle-kit writes a snapshot under $workdir/meta/0000_snapshot.json (or similar).
           // Verify exact filename by listing $workdir/meta after run; adjust path here.
           const introspected = JSON.parse(
             readFileSync(join(workdir, 'meta', '0000_snapshot.json'), 'utf-8'),

             readFileSync(
               join(__dirname, 'baseline.schema.json'),
               'utf-8',
             ),

           // Drizzle snapshots include `id` (UUID) per snapshot — strip before diff.

             return rest;
           };

         }, 60_000);
       });
