    ```typescript
    export async function startWorker(opts?: { deps?: WorkerDeps }): Promise<WorkerHandle> {
      // ...all the body currently inside main() that builds: config, redisOpts, workerConnection,
      // redisForLock, { db, pool }, dlqRepo, dlqHandler, queues, enabledWorkers, workers[],
      // cleanupQueue, heartbeat.start()...

      // the final process.exit(0)/process.exit(1) lines — surface errors by rejecting instead.
      return { shutdown: async () => { /* the cleanup body, no process.exit */ } };
    }
    ```

    Keep `main()` as a thin standalone wrapper that calls `startWorker()`, then registers the process signal handlers and calls `process.exit` on completion:

    ```typescript
    async function main() {
      const handle = await startWorker();
      let shuttingDown = false;
      const stop = async (signal: string) => {
        if (shuttingDown) return; shuttingDown = true;
        logger.info({ signal }, 'Worker shutdown initiated');
        try { await handle.shutdown(); process.exit(0); }
        catch (err) { logger.error({ err }, 'Error during worker shutdown'); process.exit(1); }
      };
      process.on('SIGTERM', () => void stop('SIGTERM'));
      process.on('SIGINT', () => void stop('SIGINT'));
    }
    main().catch((err) => { logger.error({ err }, 'Worker entrypoint failed to start'); process.exit(1); });
    ```
